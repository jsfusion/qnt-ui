import { useState, useEffect } from 'react';
import { Meta, Story, Preview, ColorPalette, ColorItem, } from '@storybook/addon-docs/blocks';
import styled from 'styled-components';
import copy from 'copy-to-clipboard';
import Icon from '../../src/components/Icon';
import { IconNames } from "@blueprintjs/icons";
import StoryTag from '../../src/internal/StoryTag';
import ContentSearch from '../../src/internal/ContentSearch';
import { Label } from '../../src';

export const IconGallery = () => {
  const icons = Object.keys(IconNames).map(name => ({ iconName: IconNames[name]}));
  const [searchTerm, setSearchTerm] = React.useState("");
  const [searchResults, setSearchResults] = React.useState([]);
  const handleChange = event => {
    setSearchTerm(event.target.value);
  };
  useEffect(() => {
    const results = icons.filter(icon =>
      icon.iconName.toLowerCase().includes(searchTerm)
    );
    setSearchResults(results);
  }, [searchTerm]);
  return (
    <>
      <ContentSearch placeholder="Search icons" isFullWidth value={searchTerm} onChange={handleChange} autocomplete="off" />
      <IconContainer>
         {searchResults.map(item => (
          <IconItem key={item.iconName} data-name={item.iconName} onClick={() => copy(item.iconName)}><Icon size={16} icon={item.iconName}/></IconItem>
        ))}
      </IconContainer>
    </>
  );
}

export const IconContainer = styled.div`
    display: flex;
    width: 100%;
    flex-wrap: wrap;
    margin-bottom: 20px;
    margin-top: 20px;
    box-shadow: inset 0 0 0 -1px '#33333f';
    cursor: pointer;
`;
